\newcommand{\rotateword}[1]{\rotatebox{180}{#1}}

\newcommand{\reverseeveryword}[1]{%
  \begingroup
  \StrCount{#1}{ }[\wordcount]
  \StrPosition[\wordcount]{#1}{ }[\spaceposition]
  \ifnum\spaceposition=0
    \rotateword{#1}%
  \else
    \StrCut{#1}{ }{\firstword}{\restofwords}%
    \rotateword{\firstword}%
    \reverseeveryword{\restofwords}%
  \fi
  \endgroup
}

\newcommand{\manju}[1]{
    \manjurm \reverseeveryword{#1}
}
